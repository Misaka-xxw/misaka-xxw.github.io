    int n;
    int f[N];
    ll res=0ll;
    int main() {
    int a;
    scanf(“%d”,&n);
    for(int i=1;i<=n;i++){
    scanf(“%d”,&a);
    f[i]=f[i-1]^a;
    }
    int bei=1;//位权
    for(int i=0;i<=20;i++){//20位二进制
    int numof1=0,numof0=1;
    ll sum=0;
    for(int j=1;j<=n;j++){
    if(f[j]&1){//最低一位1^0=1
    sum+=numof0;
    numof1++;
    }else{//最低一位0^1=1
    sum+=numof1;
    numof0++;
    }
    f[j]>>=1;//下一位
    }
